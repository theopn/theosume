% Commands for resume items

% Display four items on the far left and far right across two lines
%
% arg #1 Main item (bold large)                                 : e.g., University name, employer name
% arg #2 Item to the right of the main item (bold small)        : e.g, University location, employer location
% arg #3 Secondary item (italics large)                         : e.g., Major, specific job title
% arg #4 Item to the right of the secondary item (italics small): e.g., empolyment dates
\newcommand{\fourItems}[4]{
  \begin{tabular*}{1.0\textwidth}[t]{l@{\extracolsep{\fill}}r}
    \textbf{\large #1} & \textbf{\small #2} \\
    \textit{\large #3} & \textit{\small #4} \\
  \end{tabular*}

  \vspace*{-0.5em}
}

% Display two text items on the far left and right side.
% Useful as the extension of the \fourItems. e.g.: If you had multiple jobs under a same company
% \fourItems{Company}{Location}
%           {Latest role}{Fairly recent -- Very recent}
% - Description, description
% \twoItems{Older role}{3 years ago -- 1 year ago}
% - Description, description, description ...
%
% arg #1 Secondary item (italics large)                         : e.g., Major, specific job title
% arg #2 Item to the right of the secondary item (italics small): e.g., empolyment dates
\newcommand{\twoItems}[2]{
  \begin{tabular*}{1.0\textwidth}{l@{\extracolsep{\fill}}r}
    \textit{\large #1} & \textit{\small #2} \\
  \end{tabular*}

  \vspace*{-0.5em} % or -10pt
}
